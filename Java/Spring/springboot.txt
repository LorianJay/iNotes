@EnableSwagger2
@Configuration
public class SwaggerConfiguration {

    @Bean
    public Docket haloDefaultApi() {

        return buildApiDocket("content",
                "api.content",
                "/api/content/**");
    }

    public Docket haloAdminApi() {

        return buildApiDocket("admin",

                "/api/admin/**");

    private Docket buildApiDocket(@NonNull String groupName, @NonNull String basePackage, @NonNull String antPattern) {
        Assert.hasText(groupName, "Group name must not be blank");
        Assert.hasText(basePackage, "Base package must not be blank");
        Assert.hasText(antPattern, "Ant pattern must not be blank");

        return new Docket(DocumentationType.SWAGGER_2)
                .groupName(groupName)
                .select()
                .apis(RequestHandlerSelectors.basePackage(basePackage))
                .paths(PathSelectors.ant(antPattern))
                .build()
                .apiInfo(apiInfo());
    }

    private ApiInfo apiInfo() {
        return new ApiInfoBuilder()
                .title("API Documentation")
                .description("Documentation for API")
                .version("1.0")
                .build();
    }
}

需求:给你一个实例,返回这个实例里属性值为null的属性名称
private static Set<String> getNullPropertyNameSet(@NonNull Object source) {

    Assert.notNull(source, "source object must not be null");
    BeanWrapperImpl beanWrapper = new BeanWrapperImpl(source);				//BeanWrapperImpl,Wrapper是包装的意思相当于把我们的source包装了一层.
    PropertyDescriptor[] propertyDescriptors = beanWrapper.getPropertyDescriptors();	//获取到当前实例的所有属性名称的集合 这里会默认添加一个class属性

    Set<String> emptyNames = new HashSet<>();

    for (PropertyDescriptor propertyDescriptor : propertyDescriptors) {			//遍历属性集合
        String propertyName = propertyDescriptor.getName();				//获取当前属性的名称
        Object propertyValue = beanWrapper.getPropertyValue(propertyName);		//根据属性名称获取该属性对应指向的对象

        // if property value is equal to null, add it to empty name set
        if (propertyValue == null) {
            emptyNames.add(propertyName);			//如果这个属性没有被赋值则将属性名称添加到集合中
        }
    }
    return emptyNames;
}

在没有Token的时候,客户端每次想服务器请求时服务器都需要根据哟用户名和密码来查询数据库.Token是服务器生成的一串字符串,当第一次登陆时服务器就会生产该字符串,作为客户端访问服务器的一个令牌,以后就只需要根据这个令牌的内容来请求数据即可无需通过用户名和密码的方式来请求.

CORS跨域请求:如果现在有别的网站的部分资源引用我们的网站,相当于一个域名访问两个网站,那么真正的访问者不是别人的网站而是访问别人网站的哪个用户,也就是说访问别人网站的是用户,别人网站引用本站的资源在访问本站资源时访问者还是这个用户(所以两个网站的响应都给浏览器了).之前在学Servlet三大组件的时候学过,Filter的doFilter方法是在@Controller方法执行之前就执行的,所以这里控制所有访问网站资源的请求,通过过滤器来控制.实现跨域请求还可以通过WebMvcConfigurer的addCorsMappings方法来实现不过没有过滤器灵活.

首先在往springboot中添加过滤器Filter:
@Bean
public FilterRegistrationBean<CorsFilter> corsFilter() {
    FilterRegistrationBean<CorsFilter> corsFilter = new FilterRegistrationBean<>();

    corsFilter.setOrder(Ordered.HIGHEST_PRECEDENCE + 10);
    corsFilter.setFilter(new CorsFilter());
    corsFilter.addUrlPatterns("/api/*");

    return corsFilter;
}

public class CorsFilter extends GenericFilterBean {			//这个过滤器继承自GenericFilterBean过滤器,GenericFilterBean过滤器是spring提供的通用过滤器.

    private final static String ALLOW_HEADERS = StringUtils.joinWith(",", HttpHeaders.CONTENT_TYPE, ADMIN_TOKEN_HEADER_NAME, API_ACCESS_KEY_HEADER_NAME);

    @Override
    public void doFilter(ServletRequest request, ServletResponse response, FilterChain chain) throws IOException, ServletException {
        HttpServletRequest httpServletRequest = (HttpServletRequest) request;				//强转每什么好说的
        HttpServletResponse httpServletResponse = (HttpServletResponse) response;

        /*
        Set customized header 设置响应头
        ACCESS_CONTROL_ALLOW_ORIGIN:该值只有两种情况要么是*代表任意网站,要么是具体的网站域名.这里就是具体的网站
        域名,该域名是直接从请求参数中得到的,意思是运行请求的网站获得本站资源.
        ACCESS_CONTROL_ALLOW_HEADERS:允许请求中携带的字段
        ACCESS_CONTROL_ALLOW_METHODS:允许请求通过哪些方式请求
        ACCESS_CONTROL_ALLOW_CREDENTIALS:是否允许浏览器读取读取response的内容.
        ACCESS_CONTROL_MAX_AGE:浏览器请求本站时会分两次请求,第一次称为预检请求,第二次才是真实的请求.
        该参数对应的值是秒数,代表发送一次预检请求后间隔多少秒后的请求才会发送预检请求.如果设置为0代表每次请求都需要发送
        预检请求
         */
        httpServletResponse.setHeader(HttpHeaders.ACCESS_CONTROL_ALLOW_ORIGIN, httpServletRequest.getHeader(HttpHeaders.ORIGIN));
        httpServletResponse.setHeader(HttpHeaders.ACCESS_CONTROL_ALLOW_HEADERS, ALLOW_HEADERS);
        httpServletResponse.setHeader(HttpHeaders.ACCESS_CONTROL_ALLOW_METHODS, "GET, POST, PUT, DELETE, OPTIONS");
        httpServletResponse.setHeader(HttpHeaders.ACCESS_CONTROL_ALLOW_CREDENTIALS, "true");
        httpServletResponse.setHeader(HttpHeaders.ACCESS_CONTROL_MAX_AGE, "3600");

	/*
        判断当前请求是否是预检请求,如果是则终止过滤链,否则继续执行过滤链.
         */
        if (!CorsUtils.isPreFlightRequest(httpServletRequest)) {			//CorsUtils是springboot的工具类
            chain.doFilter(httpServletRequest, httpServletResponse);
        }
    }
}

/**
 * Controller advice for comment result.
 * 控制run.halo.app.controller包下的所有响应对象都会经过此拦截器
 * 处理正常响应的通用响应器.注意是有响应的是对象才有用(也就是响应的对象需要通过Http消息转换器转换的才会调用supports方法).
 * 例如响应的内容指向一个页面这种肯定是不过经过该响应器的.
 *
 * @author johnniang
 */
@ControllerAdvice("run.halo.app.controller")
public class CommonResultControllerAdvice implements ResponseBodyAdvice<Object> {

    @Override
    public boolean supports(MethodParameter returnType, Class<? extends HttpMessageConverter<?>> converterType) {
        // 判断哪些需要拦截,如果是AbstractJackson的消息转换器就拦截.
        return AbstractJackson2HttpMessageConverter.class.isAssignableFrom(converterType);
    }

    /**
     * 自定义修改响应头,这里主要是修改响应码
     *
     * @param body          body就是响应对象response中的响应体
     * @param returnType
     * @param contentType
     * @param converterType
     * @param request
     * @param response
     * @return
     */
    @Override
    @NonNull
    public final Object beforeBodyWrite(@Nullable Object body,
                                        MethodParameter returnType,
                                        MediaType contentType,
                                        Class<? extends HttpMessageConverter<?>> converterType,
                                        ServerHttpRequest request,
                                        ServerHttpResponse response) {
        // 根据这个body获得MappingJacksonValue,或者是将body封装在MappingJacksonValue中
        MappingJacksonValue container = getOrCreateContainer(body);
        // The contain body will never be null
        beforeBodyWriteInternal(container, contentType, returnType, request, response);
        return container;
    }

    /**
     * Wrap the body in a {@link MappingJacksonValue} value container (for providing
     * additional serialization instructions) or simply cast it if already wrapped.
     */
    private MappingJacksonValue getOrCreateContainer(Object body) {
        return (body instanceof MappingJacksonValue ? (MappingJacksonValue) body : new MappingJacksonValue(body));
    }

    private void beforeBodyWriteInternal(MappingJacksonValue bodyContainer,
                                         MediaType contentType,
                                         MethodParameter returnType,
                                         ServerHttpRequest request,
                                         ServerHttpResponse response) {
        // Get return body 获得到响应体,body.这个body就是被封装在MappingJacksonValue中的对象,也就是上面创建的对象
        Object returnBody = bodyContainer.getValue();
        /*
         有的时候在controller中会返回自定义的BaseResponse,如果这里的returnBody是BaseResponse就不使用通用的返回结果.
         */
        if (returnBody instanceof BaseResponse) {
            // If the return body is instance of BaseResponse
            BaseResponse<?> baseResponse = (BaseResponse) returnBody;
            // 根据响应码转为其对应的HttpStatus,并将响应码设置到response中
            response.setStatusCode(HttpStatus.resolve(baseResponse.getStatus()));
            return;
        }
        /*
        Wrap the return body 如果不是BaseResponse就创建一个响应码为200的实例
        并且将响应体对象returnBody封装在BaseResponse中.
         */
        BaseResponse<?> baseResponse = BaseResponse.ok(returnBody);
        bodyContainer.setValue(baseResponse);
        response.setStatusCode(HttpStatus.valueOf(baseResponse.getStatus()));
    }

}

URI themeUri = ResourceUtils.getURL("静态资源目录").toURI();
Path source;
if ("jar".equalsIgnoreCase(themeUri.getScheme())) {			//判断当前是否是以jar包的方式进行运行的
    FileSystem fileSystem = getFileSystem(themeUri);
    source = fileSystem.getPath("/BOOT-INF/classes/" + "静态资源目录");
} else {
    source = Paths.get(themeUri);
}

//此时这个path就指代我们jar包里的静态资源目录,当然静态资源目录可以指定任意文件夹.

private FileSystem getFileSystem(@NonNull URI uri) throws IOException {
    FileSystem fileSystem;

    try {
        fileSystem = FileSystems.getFileSystem(uri);
    } catch (FileSystemNotFoundException e) {
        fileSystem = FileSystems.newFileSystem(uri, Collections.emptyMap());
    }

    return fileSystem;
}

ResourceUtils是spring的一个抽象工具类:这个类主要存放一些常用的静态变量
isUrl(String) return boolean;			//判断当前这个URL格式是否正确,URL可以指代远程的资源地址也可以指代本地的一个文件 agrs0:资源地址
getURL(String) return URL;		//如果是以classpath:开头代表获取当前资源目录的文件,如果不是就创建URL如果还不是就创建File,默认得到当前项目的根目录内容. args0:资源地址

javaSystem.out.print彩色打印:
System.out.println("\033[32m test  \033[39m 时间");
通过\033[颜色编码m 来确认将要打印彩色日志

颜色编码:

ESC_START 	\033[ 	编码开始标识
BLACK_FG 	30 	黑色
RED_FG 		31 	红色
GREEN_FG 	32 	绿色
YELLOW_FG 	33 	黄色
BLUE_FG 	34 	蓝色
MAGENTA_FG 	35 	紫色
CYAN_FG 	36 	青色
WHITE_FG 	37 	白色
DEFAULT_FG 	39 	橙色
ESC_END 	m 	编码结束标识
